238 ParserState GetXML(string strLine, string& strEleName, string& strCont,
239  int nStart, int& nEnd, ParserState CurrentState) {
240 
241  // firstOpenB and firstCloseB store the spot at which the opening and
242  // closing brackets are on the line
243  int firstOpenB = strLine.find_first_of("<");
244  int firstCloseB = strLine.find_first_of(">");
245 
246  if (firstCloseB < 0) { // begins by checking if a closing tag was found
247  if (firstOpenB < 0) { //if not we are in a comment
248  xmlTemp.setContent(strLine); // sets global temp to content of the comment
249  CurrentState = IN_COMMENT;
250  return CurrentState;
251  }
252  if (strLine[firstOpenB + 1] == '!') { // if the comment is just staring
253  CurrentState = STARTING_COMMENT;
254  return CurrentState;
255  } else {
256  xmlTemp.setContent(strLine); // if the state is unknown
257  CurrentState = UNKNOWN;
258  return CurrentState;
259  }
260  } else if (firstOpenB < 0 && strLine[firstCloseB - 1] == '-') { // if the comment is ending
261  CurrentState = ENDING_COMMENT;
262  return CurrentState;
263  } else if (strLine[ firstOpenB + 1] == '?') { // if we find the directive
264  strCont = strLine;
265  strCont.erase(0, firstOpenB + 2);
266  strCont.erase(firstCloseB - 2);
267  xmlTemp.setContent(strCont); // sets global temp to content of the directive
268  CurrentState = DIRECTIVE;
269  return CurrentState;
270  } else if (strLine[ firstOpenB + 1] == '!') { // if we find a one line comment
271  strCont = strLine;
272  strCont.erase(0, firstOpenB + 4);
273  strCont.erase(firstCloseB - 6);
274  xmlTemp.setContent(strCont); // sets global temp to content of the comment
275  CurrentState = ONE_LINE_COMMENT;
276  return CurrentState;
277  } else if (strLine[ firstOpenB + 1] == '/') { // if we find a closing tag
278  CurrentState = ELEMENT_CLOSING_TAG;
279  //ShowParserState(CurrentState);
280  return CurrentState;
281  } else {
282  //takes the element name from the line
283  strEleName = strLine.substr(firstOpenB + 1, (firstCloseB - firstOpenB - 1));
284  xmlTemp.setEle(strEleName);
285

287  int lastOpenB = strLine.find_last_of("<");
288  int lastCloseB = strLine.find_last_of(">");
289  if (firstOpenB == lastOpenB && firstCloseB == lastCloseB) { // checks to see if the position of the brackets match

292  // or a self closing tag
293 
294  if (strLine[ lastCloseB - 2] == '/') { // brackets match; checks if the line is self closing
295  strCont = "{EMPTY}";
296  xmlTemp.setContent(strCont);
297  CurrentState = SELF_CLOSING_TAG;
298  //ShowParserState(CurrentState);
299  return CurrentState;
300  } else { // else it's a opening tag
301  int spaceLocation = strLine.find(" "); //finds the location of a space if there is any
302  if (spaceLocation > 0) { //function tests for spaces for the element opening tags with content within them

309  } else {
310  attrCheck = false;
311  }

319  strCont = strLine.substr(firstCloseB + 1, (lastOpenB - firstCloseB - 1));
